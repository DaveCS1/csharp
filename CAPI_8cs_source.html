981  {
982  // Create a GZIP stream with decompression mode.
983  // ... Then create a buffer and write into while reading from the GZIP stream.
984  using (GZipStream stream = new GZipStream(new MemoryStream(gzip), CompressionMode.Decompress))
985  {
986  const int size = 4096;
987  byte[] buffer = new byte[size];
988  using (MemoryStream memory = new MemoryStream())
989  {
990  int count = 0;

992  {
993  count = stream.Read(buffer, 0, size);
994  if (count > 0)
995  {
996  memory.Write(buffer, 0, count);
997  }
998  }
